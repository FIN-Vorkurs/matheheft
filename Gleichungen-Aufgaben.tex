	Für alle Aufgaben gilt grundsätzlich:
	$ x, y, z \in \mathbb{R} $ sind Variable
	$ a, b, c \in \mathbb{R} $ sind feste Parameter
	
	Lösen Sie die folgenden Gleichungen:
	\begin{enumerate}\abovedisplayskip-1em
		\item 
		\[ x^2 - x - 2 = 0 \]
		\item
		\[4x^2 + 16x - 84 = 0 \]
		\item
		\[\frac 1 2 x^2 + 3x + 4 = 0 \]
		\item
		\[4x^2 + 48x + 144 = 0 \]
		\item 
		\[(x - \sqrt{157})^2 = 0 \]
		\item
		\[\frac 7 3 x^3 + \frac {49} 3 x^2 + 35x + 21 = 0 \]
		\item
		\[\frac 7 4 x^2 + 7x = -7 \]
		\item
		\[ |x^2| = 4 \]
		\item
		\[ |x|^2 = 4 \]
		\item
		\[ |x^2 - 4 | = 2 \]
		\item
		\[x^2 = x + 12 \]
		\item
		\[3x^2 + 4x + 1 = 0\]
		\item
		\[x^5 - 25x^3 + 144x = 0 \]
		\item
		\[(x - \pi)(x + \pi) = 0 \]
		\item
		\[\frac {x^3 - 2x^2} {x - 2} + \frac {2x^2 + 4x} {x + 2} = \-1\]
		\item
		\[x^4 - 14x^3 + 59x^2 - 70x = 0\]
		\item
		\[3x^7 - 42x^5 + 147x^3 = 0\]
		\item
		\[x^{12} = 4096\]
		\item
		\[x^4 + 4x^3 + 6x^2 + 4x + 1 = 0\]
		\item
		\[(\sqrt 2 x + 2 \sqrt 2)^2 = 0\]
		\item
		\[2ax^2 - 12ax + 18a = 0\]
		\item
		\[\frac 1 {x^2} + 1 = 2\]
		\item
		\[\frac 4 x + x = 4\]
	\end{enumerate}
